<G> ::= <Assign> "сомнительно_но_окей"
<Assign> ::= <Lvalue> "я_так_чувствую" <Rvalue>

<Lvalue> ::= <Identifier>
<Rvalue> ::= <MathExpr>

<MathExpr>  ::= <AddSubRes> (("<" | "<=" | "==" | ">=" | ">" | "!=") <AddSubRes>)*
<AddSubRes> ::= <MulDivRes> (("+" | "-") <MulDivRes>)*
<MulDivRes> ::= <PowRes> (("*" | "/") <PowRes>)*
<PowRes>    ::= <Operand> ("^" <Operand>)?
<Operand>   ::= "$" <MathExpr> "$" | <SimpleOperand>

<Identifier>    ::= ([A-Z] | [a-z] | "_") ([A-Z] | [a-z] | "_" | [0-9])* " "
<Number>        ::= ([0-9])+ " "
<SimpleOperand> ::= <Identifier> | <Number>
